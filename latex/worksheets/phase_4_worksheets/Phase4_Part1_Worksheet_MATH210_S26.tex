\end{center}




%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{enumerate}[{$1.]$}]

\item Let $y(t)$ be the quantity or concentration of a substance at time $t$.  Then $y(t)$ experiences exponential growth or decay if it satisfies the following initial value problem: 
\[ \frac{dy}{dt} = ky \qquad \quad y(0) = y_0, \]
where $(t_0, y_0)$ is a known point on the graph of $y(t)$ and $k$ is the per capita growth/decay rate.  Here, $y(0)$ is called the initial data. Solve this IVP.  \vfill



\item According to Newton's Law of Cooling, the rate of change of temperature of a cooling object is proportional to the temperature \textit{difference} between the object and the surrounding medium.  That is, 
\[ \frac{dy}{dt} = -k \left( y - y_s\right), \qquad \quad y(0) = y_0 \]
where $y(t)$ is the temperature of the object at time $t$, and $y_s$ is the temperature of the surroundings, assumed to be constant.  Here, $k>0$.  Solve this IVP.  What is the equilibrium temperature? \vfill \vfill



\pagebreak 
\item Let $y(t)$ be the size of a population of interest at time $t$.  Given limited resources or space to expand, many populations experience \textit{logistic} growth at a per capita rate $k>0$ to a \textit{carrying capacity}, $m$:

\[ \frac{dy}{dt} =ky \left( 1 - \frac{y}{m}\right), \qquad \quad y(0) = y_0 .\]

Here, $k$ and $m$ are both strictly positive.  Solve this ODE for $y(t)$ and determine the equilibria. 

\end{enumerate}
