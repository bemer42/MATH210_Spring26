\Large{\textsc{Phase 1 (part 1): Critical Points and Relative Extrema}}\\
\end{center}

\begin{enumerate}[{$1.]$}]

\item Find all the critical points of the following functions on the given intervals:
\begin{enumerate}
	\item $\displaystyle f(x) = 3x^4 + 8x^3 - 18x^2$ on $[-1, 2]$\vfill
	
	\item $\displaystyle g(x) = x^{1/3}(x-2)^2$ on $(-\infty, \infty)$\vfill\vfill
\end{enumerate}


\pagebreak

\item Determine and classify all relative extrema of the the following functions: 
\begin{enumerate}
	\item $\displaystyle f(x) = \frac{2}{3}x^3 + 4x^2 - 10x + \frac{5}{3}$ \vfill
	\item $\displaystyle g(x) = x^2 e^{-x/2}$ \vfill
	
\end{enumerate}



\end{enumerate}
